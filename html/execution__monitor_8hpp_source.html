191  {
192  }
193 
194  // Destructor
195  virtual ~translator_holder_base() {}
196 
197  // translator holder interface
198  // invokes the function F inside the try/catch guarding against specific exception
199  virtual int operator()( boost::function<int ()> const& F ) = 0;
200 
201  // erases specific translator holder from the chain
202  translator_holder_base_ptr erase( translator_holder_base_ptr this_, const_string tag )
203  {
204  if( m_next )
205  m_next = m_next->erase( m_next, tag );
206 
207  return m_tag == tag ? m_next : this_;
208  }
209 #ifndef BOOST_NO_RTTI
210  virtual translator_holder_base_ptr erase( translator_holder_base_ptr this_, std::type_info const& ) = 0;
211  template<typename ExceptionType>
212  translator_holder_base_ptr erase( translator_holder_base_ptr this_, boost::type<ExceptionType>* = 0 )
213  {
214  if( m_next )
215  m_next = m_next->erase<ExceptionType>( m_next );
216 
217  return erase( this_, typeid(ExceptionType) );
218  }
219 #endif
220 
221 protected:
222  // Data members
223  translator_holder_base_ptr m_next;
224  std::string m_tag;

435 template<typename ExceptionType, typename ExceptionTranslator>
436 class translator_holder : public translator_holder_base
437 {
438 public:
439  explicit translator_holder( ExceptionTranslator const& tr, translator_holder_base_ptr& next, const_string tag = const_string() )
440  : translator_holder_base( next, tag ), m_translator( tr ) {}
441 
442  // translator holder interface
443  virtual int operator()( boost::function<int ()> const& F )
444  {
445  BOOST_TEST_I_TRY {
446  return m_next ? (*m_next)( F ) : F();
447  }
448  BOOST_TEST_I_CATCH( ExceptionType, e ) {
449  m_translator( e );
450  return boost::exit_exception_failure;
451  }
452  }
453 #ifndef BOOST_NO_RTTI
454  virtual translator_holder_base_ptr erase( translator_holder_base_ptr this_, std::type_info const& ti )
455  {
456  return ti == typeid(ExceptionType) ? m_next : this_;
457  }
458 #endif
459 
460 private:
461  // Data members
462  ExceptionTranslator m_translator;
463 };
464 
465 } // namespace detail
