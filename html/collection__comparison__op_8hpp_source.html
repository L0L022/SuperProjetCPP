54 template <typename OP, bool can_be_equal, bool prefer_shorter,
55  typename Lhs, typename Rhs>
56 inline
57 typename boost::enable_if_c<
58  unit_test::is_forward_iterable<Lhs>::value && unit_test::is_forward_iterable<Rhs>::value,
59  assertion_result>::type
60 lexicographic_compare( Lhs const& lhs, Rhs const& rhs )
61 {
62  assertion_result ar( true );
63 
64  typename Lhs::const_iterator first1 = lhs.begin();
65  typename Rhs::const_iterator first2 = rhs.begin();
66  typename Lhs::const_iterator last1 = lhs.end();
67  typename Rhs::const_iterator last2 = rhs.end();
68  std::size_t pos = 0;
69 
70  for( ; (first1 != last1) && (first2 != last2); ++first1, ++first2, ++pos ) {
71  assertion_result const& element_ar = OP::eval(*first1, *first2);
72  if( !can_be_equal && element_ar )
73  return ar; // a < b
74 
75  assertion_result const& reverse_ar = OP::eval(*first2, *first1);
76  if( element_ar && !reverse_ar )
77  return ar; // a<=b and !(b<=a) => a < b => return true
78 
79  if( element_ar || !reverse_ar )
80  continue; // (a<=b and b<=a) or (!(a<b) and !(b<a)) => a == b => keep looking
81 
82  // !(a<=b) and b<=a => b < a => return false
83  ar = false;
84  ar.message() << "\nFailure at position " << pos << ": "
85  << tt_detail::print_helper(*first1)
86  << OP::revert()
87  << tt_detail::print_helper(*first2)
88  << ". " << element_ar.message();
89  return ar;
90  }
91 
92  if( first1 != last1 ) {
93  if( prefer_shorter ) {
94  ar = false;
95  ar.message() << "\nFirst collection has extra trailing elements.";
96  }
97  }
98  else if( first2 != last2 ) {
99  if( !prefer_shorter ) {
100  ar = false;
101  ar.message() << "\nSecond collection has extra trailing elements.";
102  }
103  }
104  else if( !can_be_equal ) {
105  ar = false;
106  ar.message() << "\nCollections appear to be equal.";
107  }
108 
109  return ar;
110 }

135 template <typename OP, typename Lhs, typename Rhs>
136 inline
137 typename boost::enable_if_c<
138  unit_test::is_forward_iterable<Lhs>::value && unit_test::is_forward_iterable<Rhs>::value,
139  assertion_result>::type
140 element_compare( Lhs const& lhs, Rhs const& rhs )
141 {
142  assertion_result ar( true );
143 
144  if( lhs.size() != rhs.size() ) {
145  ar = false;
146  ar.message() << "\nCollections size mismatch: " << lhs.size() << " != " << rhs.size();
147  return ar;
148  }
149 
150  typename Lhs::const_iterator left = lhs.begin();
151  typename Rhs::const_iterator right = rhs.begin();
152  std::size_t pos = 0;
153 
154  for( ; pos < lhs.size(); ++left, ++right, ++pos ) {
155  assertion_result const element_ar = OP::eval( *left, *right );
156  if( element_ar )
157  continue;
158 
159  ar = false;
160  ar.message() << "\nMismatch at position " << pos << ": "
161  << tt_detail::print_helper(*left)
162  << OP::revert()
163  << tt_detail::print_helper(*right)
164  << ". " << element_ar.message();
165  }
166 
167  return ar;
168 }

192 template <typename OP, typename Lhs, typename Rhs>
193 inline assertion_result
194 non_equality_compare( Lhs const& lhs, Rhs const& rhs )
195 {
196  assertion_result ar( true );
197 
198  if( lhs.size() != rhs.size() )
199  return ar;
200 
201  typename Lhs::const_iterator left = lhs.begin();
202  typename Rhs::const_iterator right = rhs.begin();
203  typename Lhs::const_iterator end = lhs.end();
204 
205  for( ; left != end; ++left, ++right ) {
206  if( OP::eval( *left, *right ) )
207  return ar;
208  }
209 
210  ar = false;
211  ar.message() << "\nCollections appear to be equal";
212 
213  return ar;
214 }

367 #define DEFINE_COLLECTION_COMPARISON( oper, name, rev ) \
368 template<typename Lhs,typename Rhs> \
369 struct name<Lhs,Rhs,typename boost::enable_if_c< \
370  unit_test::is_forward_iterable<Lhs>::value \
371  && !unit_test::is_cstring<Lhs>::value \
372  && unit_test::is_forward_iterable<Rhs>::value \
373  && !unit_test::is_cstring<Rhs>::value>::type> { \
374 public: \
375  typedef assertion_result result_type; \
376  \
377  typedef name<Lhs, Rhs> OP; \
378  typedef typename \
379  mpl::if_c<is_same<typename decay<Lhs>::type, \
380  typename decay<Rhs>::type>::value, \
381  typename cctraits<OP>::is_specialized, \
382  mpl::false_>::type is_specialized; \
383  \
384  typedef name<typename Lhs::value_type, \
385  typename Rhs::value_type> elem_op; \
386  \
387  static assertion_result \
388  eval( Lhs const& lhs, Rhs const& rhs) \
389  { \
390  return assertion::op::compare_collections( lhs, rhs, \
391  (boost::type<elem_op>*)0, \
392  is_specialized() ); \
393  } \
394  \
395  template<typename PrevExprType> \
396  static void \
397  report( std::ostream&, \
398  PrevExprType const&, \
399  Rhs const& ) {} \
400  \
401  static char const* revert() \
402  { return " " #rev " "; } \
403  \
404 }; \
